SELECT AVG(PURCHASE_AMOUNT) AS "AVG_PURCHASE" FROM ORDERS;
SELECT MAX(PURCHASE_AMOUNT) AS "MAX_PURCHASE" FROM ORDERS;
SELECT MIN(PURCHASE_AMOUNT) AS "MIN_PURCHASE"  FROM ORDERS;
